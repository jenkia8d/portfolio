WITH cte AS (
    SELECT 
    date_format(report_time, 'MM-dd HH:mm:ss') AS report_time_formatted,
    array_index, 
    string_index,
    concat(array_index, '_', string_index) AS entity,
    CASE
        WHEN in_rotation = TRUE THEN 'IN'
        ELSE 'OUT'
    END AS rotation_switch_to,
    CASE 
        WHEN LAG(in_rotation) OVER (PARTITION BY array_index, string_index ORDER BY report_time) <> in_rotation THEN TRUE
        ELSE FALSE
    END AS rotation_state_changed,

    CASE
        WHEN positive_contactor_closed = TRUE THEN 'CLOSED'
        ELSE 'OPEN'
    END AS pos_contactor_switch_to,
    CASE 
        WHEN LAG(positive_contactor_closed) OVER (PARTITION BY array_index, string_index ORDER BY report_time) <> positive_contactor_closed THEN TRUE
        ELSE FALSE
    END AS pos_contactor_changed
    
    FROM string_report_view
    WHERE station_code = 'INV0111'
    AND block_index = 1
    -- AND report_time BETWEEN '2025-06-24 18:40:00' AND '2025-06-24 18:50:00'
    AND date(report_time) = '2025-06-24'
)
SELECT report_time_formatted, array_index, string_index, entity, rotation_state_changed, rotation_switch_to, pos_contactor_changed, pos_contactor_switch_to
FROM cte
WHERE ((rotation_state_changed = TRUE) OR (pos_contactor_changed = TRUE))
ORDER BY report_time_formatted, array_index, string_index
